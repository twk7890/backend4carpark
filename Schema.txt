Schema

Place = new Mongo.Collection("place");
	Place.attachSchema(new SimpleSchema({
		userId: {
			type: String
		},
		region: {
			type: String
		},
		address: {
			type: String
		},
		timezone: {
			type: Date
		},
		state: {
			type: String,
			autoform: {
			      type: "select",
			      options: function () {
			        return [
			          {label: "active", value: "ACTIVE"},
			          {label: "suspend", value: "SUSPEND"},
			        ];
			      }
			    }
		},
		createdAt: {
			type: Date,
			autoform.omit:true,
			autoValue:function(){
				return this.isInsert? new Date() : this.val;
			}
		}
	}));
